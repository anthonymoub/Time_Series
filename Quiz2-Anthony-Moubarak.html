library(shiny)
library(quantmod)
library(plotly)

# Define UI for the Shiny app
ui <- fluidPage(
  titlePanel("Stock Price Line Plot"),
  
  sidebarLayout(
    sidebarPanel(
      selectInput("stock_symbols_1", "Select S&P500 stocks",
                  choices = sp500_symbols,
                  selected = c("AAPL"),
                  multiple = TRUE),
      selectInput("stock_symbols_2", "Select Dow Jones stocks",
                  choices = djia_symbols,
                  selected = c("CSCO"),
                  multiple = TRUE),
      dateInput("start_date", "Start Date:", value = Sys.Date() - 365),
      dateInput("end_date", "End Date:", value = Sys.Date())
    ),
    
    mainPanel(
      plotlyOutput("stock_plot_1"),
      plotlyOutput("stock_plot_2")
    )
  )
)

# Define server logic
server <- function(input, output) {
  
  # Function to retrieve stock data
  getStockData <- reactive({
    symbols_1 <- toupper(input$stock_symbols_1)
    symbols_2 <- toupper(input$stock_symbols_2)
    start_date <- input$start_date
    end_date <- input$end_date
    
    stock_data <- NULL
    
    for (symbol in symbols_1) {
      stock <- getSymbols(symbol, from = start_date, to = end_date, auto.assign = FALSE)
      if (!is.null(stock)) {
        cc <- Cl(stock)
        colnames(cc) <- c("Price")
        if (is.null(stock_data)) {
          stock_data <- data.frame(Date = index(stock), Price = cc, Symbol = symbol)
        } else {
          stock_data <- rbind(stock_data, data.frame(Date = index(stock), Price = cc, Symbol = symbol))
        }
      }
    }
    
    return(stock_data)
  })
  
  # Function to retrieve stock data for the second plot
  getStockData2 <- reactive({
    symbols_2 <- toupper(input$stock_symbols_2)
    start_date <- input$start_date
    end_date <- input$end_date
    
    stock_data_2 <- NULL
    
    for (symbol in symbols_2) {
      stock <- getSymbols(symbol, from = start_date, to = end_date, auto.assign = FALSE)
      if (!is.null(stock)) {
        cc <- Cl(stock)
        colnames(cc) <- c("Price")
        if (is.null(stock_data_2)) {
          stock_data_2 <- data.frame(Date = index(stock), Price = cc, Symbol = symbol)
        } else {
          stock_data_2 <- rbind(stock_data_2, data.frame(Date = index(stock), Price = cc, Symbol = symbol))
        }
      }
    }
    
    return(stock_data_2)
  })
  
  # Render the first stock price line plot
  output$stock_plot_1 <- renderPlotly({
    stock_data <- getStockData()
    
    if (!is.null(stock_data)) {
      p <- plot_ly(data = stock_data, x = ~Date, y = ~Price, type = 'scatter', mode = 'lines', color = ~Symbol)
      
      p <- p %>% layout(title = "Stock Prices (S&P500 stocks)",
                        xaxis = list(rangeslider = list(visible = TRUE), title = "Date"),
                        yaxis = list(title = "Price"))
      
      p
    }
  })
  
  # Render the second stock price line plot
  output$stock_plot_2 <- renderPlotly({
    stock_data_2 <- getStockData2()
    
    if (!is.null(stock_data_2)) {
      p <- plot_ly(data = stock_data_2, x = ~Date, y = ~Price, type = 'scatter', mode = 'lines', color = ~Symbol)
      
      p <- p %>% layout(title = "Stock Prices (Dow Jones stocks)",
                        xaxis = list(rangeslider = list(visible = TRUE), title = "Date"),
                        yaxis = list(title = "Price"))
      
      p
    }
  })
}

I selected three stocks from the S&P 500 index: AAPL, AMZN, and GOOGL, with the aim of analyzing their performance from the beginning of 2019 until now. For the Dow Jones index, I chose CSCO, IBM, and MSFT. The reason for this selection was to compare the tech industry’s performance across both indices, considering that the tech sector has been one of the fastest-growing in the market in recent years.

At first glance, it’s evident that the S&P 500 stocks have exhibited greater volatility, resulting in less smooth stock price trajectories compared to those in the Dow Jones. However, all three S&P 500 stocks ended up with significantly higher prices by the end of 2023 compared to January 2019. Additionally, by using the plot’s zoom-in feature, we can observe that all three stocks experienced significant increases in the first half of 2023.

An article published on Fool.com titled “Why Tech Stocks Jumped in the First Half of 2023” provides insights into the reasons behind this trend. The main factors seem to be the market’s recovery from a dip and increased investor confidence in the tech stock market, given its rapid growth potential. Furthermore, recent tech layoffs may have signaled higher end-of-year profits, contributing to the spike and recovery from the COVID-19 pandemic.

Conversely, the Dow Jones-selected stocks have shown lower volatility, with smoother price trajectories, especially in the cases of CSCO and IBM. What’s interesting is that CSCO and IBM, which can be considered part of the same cluster, apart from Microsoft, have followed a similar steady pattern and have seen minimal price increases in the past couple of years.

An article titled “Is the Telecoms Industry Really in Freefall?” explains the slowdown in the telecom industry’s stock market growth. Factors such as increased competition, the commoditization of core services, demands for greater infrastructure investment, and pressure from big tech companies to share network maintenance and expansion costs have contributed to this trend.

Question 1

Stationarity in a time series means that the series’ statistical properties do not change over time. In a stationary time series, the mean, variance, and autocorrelation structure remain constant over time. Specifically:

  • Constant Mean: The mean of the series should not be a function of time. It should be constant.
  • Constant Variance: The variance of the series should not change over time, ensuring that the series does not show periods of high and low volatility.
  • Constant Autocorrelation Structure:The autocorrelation between the values of the series at different points in time should only depend on the time difference between the two points, not on the original time at which the points were measured.

library(dplyr)
library(ggplot2)
library(scales)

# Create a new column month_year to group by
df <- df %>% mutate(month_year = format(date, "%b_%Y"))

# Group by month_year and then summarize
df_aggregated <- df %>% 
  group_by(month_year) %>% 
  summarise(
    cardiovasc_death_rate = mean(cardiovasc_death_rate, na.rm = TRUE),
    people_fully_vaccinated_per_hundred = mean(people_fully_vaccinated_per_hundred, na.rm = TRUE)
  ) %>% 
  ungroup()

# Plotting
p <- ggplot(df_aggregated, aes(x = as.Date(paste(month_year, "01", sep = "_"), format = "%b_%Y_%d"))) + 
  geom_line(aes(y = cardiovasc_death_rate), color = "red", na.rm = TRUE) + 
  geom_line(aes(y = people_fully_vaccinated_per_hundred), color = "blue", na.rm = TRUE, linetype="twodash") +
  scale_y_continuous(
    "Cardiovascular Death Rate", 
    sec.axis = sec_axis(~., name = "People Fully Vaccinated per Hundred")
  ) +
  labs(
    title = "Worldwide Cardiovascular Death Rate and Vaccination Rate over Time",
    color = "Metric"
  ) +
  theme_minimal()
