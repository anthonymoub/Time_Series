import numpy as np

# Assuming df is your DataFrame with columns 'weekend' and 'top10_gross'
# Convert 'top10_gross' to a NumPy array
data = df['top10_gross'].values

# Define the sequence length (e.g., 10)
sequence_length = 10

# Create sequences and corresponding labels
sequences = []
labels = []

for i in range(len(data) - sequence_length):
    seq = data[i:i+sequence_length]
    label = data[i+sequence_length]
    sequences.append(seq)
    labels.append(label)

# Convert to NumPy arrays
X = np.array(sequences)
y = np.array(labels)

# Keep the last 100 entries as a validation set
X_val = X[-100:]
y_val = y[-100:]

# Remove the validation set from the training data
X = X[:-100]
y = y[:-100]

# Reshape X to fit the (batch_size, sequence_length, num_features) format
X = X.reshape((X.shape[0], X.shape[1], 1))
X_val = X_val.reshape((X_val.shape[0], X_val.shape[1], 1))

import pandas as pd
from statsmodels.tsa.arima.model import ARIMA
from math import sqrt

# Load your DataFrame
# df = ... # Load your DataFrame here

# Assuming df['top10_gross'] contains the time series data
data = df['top10_gross']

# Leave out the last 100 observations for final validation
train_data = data[:-100]

# Number of observations in the training data
n = len(train_data)

# Number of folds for cross-validation
n_folds = 5

# Calculate the size of each fold
fold_size = n // n_folds

# Perform 5-fold cross-validation
for i in range(n_folds):
    # Define the end of the training set
    train_end = i * fold_size + fold_size

    # Make sure we don't go beyond the dataset
    if train_end >= n:
        break

    # Define the training and test data for each fold
    train = train_data.iloc[:train_end]
    test = train_data.iloc[train_end:train_end + fold_size]

    # Fit the ARIMA model
    model = ARIMA(train, order=(3, 1, 2))
    model_fit = model.fit()

    # Evaluate the model on the test set
    # Ensure there are enough data points in the test set
    if len(test) > 0:
        predictions = model_fit.forecast(steps=len(test))
        # Calculate RMSE for the test set
        test_rmse = sqrt(mean_squared_error(test, predictions))

        # Calculate RMSE for the training set
        train_predictions = model_fit.predict(start=0, end=len(train)-1)
        train_rmse = sqrt(mean_squared_error(train, train_predictions))

        # Print model summary and evaluation metrics
        print(f'Fold {i + 1}')

        print(f'Train RMSE: {train_rmse}')
        print(f'Test RMSE: {test_rmse}')
    else:
        print(f"Skipping fold {i + 1} due to insufficient test data")
